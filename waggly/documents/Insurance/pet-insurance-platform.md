```typescript
function calculateInsuranceScore(pet, plan) {
  let score = 0;
  
  // Breed-specific risks
  if (pet.breed === 'Golden Retriever') {
    if (plan.coversChronic && plan.coversOrthopedic) {
      score += 20; // Hip dysplasia common
    }
  }
  
  // Age factor
  if (pet.age < 2) {
    if (plan.accidentCoverage) score += 15; // Puppies = accidents
  } else if (pet.age > 8) {
    if (plan.noAgeLimit && plan.coversCancer) {
      score += 20; // Senior dogs need comprehensive
    }
  }
  
  // Health history
  if (pet.preExistingConditions.length === 0) {
    score += 10; // Healthy = more plan options
  }
  
  // Cost to coverage ratio
  const valueScore = (plan.coverage / plan.price) * 10;
  score += valueScore;
  
  // User reviews
  score += plan.rating * 4; // 4.5 stars = 18 points
  
  // Waggli community usage
  if (plan.waggliUsers > 500) score += 10;
  
  return score;
}
```

```sql
-- Insurance Providers
CREATE TABLE insurance_providers (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name VARCHAR(100) NOT NULL,
  logo_url TEXT,
  website_url TEXT,
  affiliate_link_template TEXT,
  
  -- Commission details
  commission_type VARCHAR(20), -- 'cpl', 'cps', 'revenue_share'
  commission_amount DECIMAL(10,2),
  commission_percentage DECIMAL(5,2),
  cookie_duration_days INTEGER,
  
  -- API Integration
  api_enabled BOOLEAN DEFAULT false,
  api_key_encrypted TEXT,
  api_endpoint TEXT,
  
  -- Ratings
  overall_rating DECIMAL(3,2),
  total_reviews INTEGER DEFAULT 0,
  
  -- Metadata
  countries TEXT[], -- ['US', 'UK', 'UAE']
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Insurance Quotes (Cache)
CREATE TABLE insurance_quotes (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  pet_id UUID REFERENCES pets(id) ON DELETE CASCADE,
  
  -- Quote request data
  coverage_type VARCHAR(50), -- 'accident_illness', 'accident_only'
  deductible_amount DECIMAL(10,2),
  reimbursement_rate INTEGER, -- 70, 80, 90, 100
  annual_limit DECIMAL(10,2), -- 0 = unlimited
  
  -- Quote results (JSONB for flexibility)
  quotes JSONB, -- Array of provider quotes
  
  -- Metadata
  zip_code VARCHAR(10),
  requested_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  expires_at TIMESTAMP WITH TIME ZONE,
  
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE INDEX idx_insurance_quotes_user ON insurance_quotes(user_id);
CREATE INDEX idx_insurance_quotes_pet ON insurance_quotes(pet_id);

-- Insurance Applications (Referral Tracking)
CREATE TABLE insurance_applications (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES auth.users(id),
  pet_id UUID REFERENCES pets(id),
  provider_id UUID REFERENCES insurance_providers(id),
  
  -- Application details
  plan_name VARCHAR(200),
  monthly_premium DECIMAL(10,2),
  annual_premium DECIMAL(10,2),
  
  -- Referral tracking
  affiliate_click_id VARCHAR(200), -- Provider's tracking ID
  referral_url TEXT,
  
  -- Status tracking
  status VARCHAR(50), -- 'clicked', 'quote_requested', 'application_started', 'policy_purchased', 'cancelled'
  
  -- Commission tracking
  commission_earned DECIMAL(10,2),
  commission_status VARCHAR(50), -- 'pending', 'approved', 'paid'
  commission_paid_at TIMESTAMP WITH TIME ZONE,
  
  -- Timestamps
  clicked_at TIMESTAMP WITH TIME ZONE,
  quote_requested_at TIMESTAMP WITH TIME ZONE,
  application_started_at TIMESTAMP WITH TIME ZONE,
  policy_purchased_at TIMESTAMP WITH TIME ZONE,
  
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE INDEX idx_insurance_applications_user ON insurance_applications(user_id);
CREATE INDEX idx_insurance_applications_status ON insurance_applications(status);
CREATE INDEX idx_insurance_applications_commission ON insurance_applications(commission_status) 
  WHERE commission_status = 'pending';

-- Insurance Reviews
CREATE TABLE insurance_reviews (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  provider_id UUID REFERENCES insurance_providers(id),
  application_id UUID REFERENCES insurance_applications(id),
  
  -- Review content
  rating INTEGER NOT NULL CHECK (rating BETWEEN 1 AND 5),
  title VARCHAR(200),
  review_text TEXT,
  
  -- Secondary ratings
  claims_rating INTEGER CHECK (claims_rating BETWEEN 1 AND 5),
  service_rating INTEGER CHECK (service_rating BETWEEN 1 AND 5),
  value_rating INTEGER CHECK (value_rating BETWEEN 1 AND 5),
  
  -- Verification
  verified_purchase BOOLEAN DEFAULT true,
  months_with_provider INTEGER,
  
  -- Engagement
  helpful_count INTEGER DEFAULT 0,
  not_helpful_count INTEGER DEFAULT 0,
  
  -- Moderation
  is_approved BOOLEAN DEFAULT false,
  is_featured BOOLEAN DEFAULT false,
  
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE INDEX idx_insurance_reviews_provider ON insurance_reviews(provider_id);
CREATE INDEX idx_insurance_reviews_rating ON insurance_reviews(rating);
CREATE INDEX idx_insurance_reviews_verified ON insurance_reviews(verified_purchase) 
  WHERE verified_purchase = true;

-- Insurance Review Votes
CREATE TABLE insurance_review_votes (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  review_id UUID REFERENCES insurance_reviews(id) ON DELETE CASCADE,
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  vote_type VARCHAR(20) NOT NULL, -- 'helpful', 'not_helpful'
  
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  
  UNIQUE(review_id, user_id)
);
```


4. Redirect user to provider website
5. Track click event for analytics

**Process**:
1. Verify webhook signature (security)
2. Find `insurance_applications` record by `affiliate_click_id`
3. Update status to 'policy_purchased'
4. Record commission amount
5. Send congratulations email to user
6. Update analytics

### B. Glossary

**Accident-Only**: Covers only injuries from accidents (broken bones, poisoning, etc.)  
**Accident & Illness**: Comprehensive coverage for accidents and illnesses  
**Annual Limit**: Maximum payout per year  
**Deductible**: Amount you pay before insurance kicks in  
**Pre-existing Condition**: Health issues that existed before policy started (usually not covered)  
**Reimbursement Rate**: Percentage insurance covers (70%, 80%, 90%, 100%)  
**Waiting Period**: Time between policy start and when coverage begins (usually 14 days)  
**Wellness Coverage**: Optional add-on for routine care (vaccines, checkups)
